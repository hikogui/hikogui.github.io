381  [[nodiscard]] friend axis_aligned_rectangle
382  align(axis_aligned_rectangle haystack, axis_aligned_rectangle needle, alignment alignment) noexcept
383  {
384  T x;
385  if (alignment == horizontal_alignment::left) {
386  x = haystack.p0().x();
387 
388  } else if (alignment == horizontal_alignment::right) {
389  x = haystack.p3().x() - needle.width();
390 
391  } else if (alignment == horizontal_alignment::center) {
392  x = (haystack.p0().x() + (haystack.width() * 0.5f)) - (needle.width() * 0.5f);
393 
394  } else {
395  tt_no_default();
396  }
397 
398  T y;
399  if (alignment == vertical_alignment::bottom) {
400  y = haystack.p0().y();
401 
402  } else if (alignment == vertical_alignment::top) {
403  y = haystack.p3().y() - needle.height();
404 
405  } else if (alignment == vertical_alignment::middle) {
406  y = (haystack.p0().y() + (haystack.height() * 0.5f)) - (needle.height() * 0.5f);
407 
408  } else {
409  tt_no_default();
410  }
411 
412  return {numeric_array<T, 4>::point({x, y}), static_cast<f32x4>(needle.extent())};
413  }

433  [[nodiscard]] friend bool overlaps(axis_aligned_rectangle const &lhs, axis_aligned_rectangle const &rhs) noexcept
434  {
435  if (lhs.empty() || rhs.empty()) {
436  return false;
437  }
438 
439  ttlet rhs_swap = rhs.v.zwxy();
440 
441  // lhs.p0.x > rhs.p3.x | lhs.p0.y > rhs.p3.y
442  if ((gt(lhs.v, rhs_swap) & 0b0011) != 0) {
443  return false;
444  }
445 
446  // lhs.p3.x < rhs.p0.x | lhs.p3.y < rhs.p0.y
447  if ((lt(lhs.v, rhs_swap) & 0b1100) != 0) {
448  return false;
449  }
450 
451  return true;
452  }
