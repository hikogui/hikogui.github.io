39hi_inline std::tuple<bool, std::size_t, bool, std::size_t> _parse_spreadsheet_address(std::string_view& address)
40{
41 bool column_nr_is_relative = true;
42 std::size_t column_nr = 0;
43 bool row_nr_is_relative = true;
44 std::size_t row_nr = 0;
45
46 if (address.starts_with("$")) {
47 column_nr_is_relative = false;
48 address.remove_prefix(1);
49 }
50
51 while (!address.empty() && is_alpha(address.front())) {
52 column_nr *= 26;
53 column_nr += address.front() - (is_upper(address.front()) ? 'A' : 'a') + 1;
54 address.remove_prefix(1);
55 }
56
57 if (address.starts_with("$")) {
58 row_nr_is_relative = false;
59 address.remove_prefix(1);
60 }
61
62 while (!address.empty() && is_digit(address.front())) {
63 row_nr *= 10;
64 row_nr += address.front() - '0';
65 address.remove_prefix(1);
66 }
67
68 return {column_nr_is_relative, column_nr - 1, row_nr_is_relative, row_nr - 1};
69}
