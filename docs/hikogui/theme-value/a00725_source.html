95 [[nodiscard]] lean_vector<glyph_id> find_glyph(grapheme g) const
96 {
97 // Create a glyph_ids object for a single grapheme.
98 auto r = lean_vector<glyph_id>{};
99
100 // First try composed normalization
101 for (auto const c : g.composed()) {
102 if (auto const glyph_id = find_glyph(c)) {
103 r.push_back(glyph_id);
104 } else {
105 r.clear();
106 break;
107 }
108 }
109
110 if (r.empty()) {
111 // Now try decomposed normalization
112 for (auto const c : g.decomposed()) {
113 if (auto const glyph_id = find_glyph(c)) {
114 r.push_back(glyph_id);
115 } else {
116 r.clear();
117 break;
118 }
119 }
120 }
121
122 return r;
123 }
