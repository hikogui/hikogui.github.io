19template<typename R, typename... Args>
20struct callback_base {
21 virtual R operator()(Args... args) = 0;
22
25 [[nodiscard]] bool expired() const noexcept
26 {
27 return (count.load(std::memory_order::relaxed) & 1) == 0;
28 }
29
32 [[nodiscard]] bool locked() const noexcept
33 {
34 return count.load(std::memory_order::relaxed) >= 2;
35 }
36
45 [[nodiscard]] bool lock() const noexcept
46 {
47 auto expected = count.load(std::memory_order::relaxed);
48 do {
49 hi_axiom(expected < std::numeric_limits<std::atomic_unsigned_lock_free::value_type>::max() - 2);
50
51 if ((expected & 1) == 0) {
52 // The callback object is being/was destructed.
53 return false;
54 }
55 } while (not count.compare_exchange_strong(expected, expected + 2, std::memory_order::relaxed));
56
57 return true;
58 }
59
62 void unlock() const noexcept
63 {
64 if (count.fetch_sub(2, std::memory_order::relaxed) == 2) {
65 // Last inflight callback has finished,
66 // and callback object is being destroyed.
67 // Notify a callback object that is delaying destruction.
68 count.notify_one();
69 }
70 }
71
74 void unsafe_destruct() const noexcept
75 {
76 hilet prev_count = count.fetch_sub(1, std::memory_order::relaxed);
77 hi_axiom((prev_count & 1) == 1);
78 }
79
82 void delay_destruct() const noexcept
83 {
84 if (auto expected = count.fetch_sub(1, std::memory_order::relaxed); expected != 0) {
85 // There are callbacks in-flight, wait for any callbacks to finish.
86 do {
87 count.wait(expected, std::memory_order::relaxed);
88 expected = count.load(std::memory_order::relaxed);
89
90 hi_axiom((expected & 1) == 0);
91 } while (expected != 0);
92 }
93 }
94
101 mutable std::atomic_unsigned_lock_free count = 1;
102};
