116[[nodiscard]] hi_inline std::optional<std::filesystem::path> source_dir_parse_cmake_install(std::filesystem::path path) noexcept
117{
118 if (not std::filesystem::exists(path)) {
119 return std::nullopt;
120 }
121
122 auto line = std::string{};
123 try {
124 auto fd = std::ifstream{path.string()};
125 line = getline(fd, 512);
126 fd.close();
127
128 } catch (...) {
129 return std::nullopt;
130 }
131
132 hilet cmake_install_start = std::string{"# Install script for directory: "};
133 if (not line.starts_with(cmake_install_start)) {
134 return std::nullopt;
135 }
136
137 auto source_dir = std::filesystem::path{line.substr(cmake_install_start.size())};
138 if (not std::filesystem::exists(source_dir)) {
139 return std::nullopt;
140 }
141
142 return source_dir;
143}

151[[nodiscard]] hi_inline std::optional<std::filesystem::path> source_dir() noexcept
152{
153 // If the cmake_install.cmake file exists then the executable is located in a build directory.
154 if (auto path = source_dir_parse_cmake_install(executable_dir() / "cmake_install.cmake")) {
155 return *path;
156 }
157
158 // When using a cmake multi-config generator, the cmake_install.cmake file is located one directory up.
159 if (auto path = source_dir_parse_cmake_install(executable_dir() / ".." / "cmake_install.cmake")) {
160 return *path;
161 }
162
163 return std::nullopt;
164}

172[[nodiscard]] hi_inline std::optional<std::filesystem::path> library_install_dir() noexcept
173{
174 // path is:
175 // - /install_dir/include/hikogui/path/path_location_impl.hpp
176 // - /build_dir/src/hikogui/path/path_location_impl.hpp
177 // becomes:
178 // - /install_dir/
179 // - /build_dir/
180 auto path = std::filesystem::path{__FILE__};
181 path.replace_filename("../../..");
182 auto install_path = std::filesystem::canonical(path);
183
184 if (install_path != library_source_dir()) {
185 return install_path;
186 } else {
187 return std::nullopt;
188 }
189}
