51 widget_intf(widget_intf const *parent) noexcept :
52 id(make_widget_id()), parent(const_cast<widget_intf *>(parent))
53 {
54 // This lambda allows the state to be set once before it will trigger
55 // notifications.
56 _state_cbt = state.subscribe([&](widget_state new_state) {
57 static std::optional<widget_state> old_state = std::nullopt;
58
59 if (old_state) {
60 if (need_reconstrain(*old_state, *state)) {
61 ++global_counter<"widget:state:reconstrain">;
62 process_event({gui_event_type::window_reconstrain});
63
64 } else if (need_relayout(*old_state, *state)) {
65 ++global_counter<"widget:state:relayout">;
66 process_event({gui_event_type::window_relayout});
67
68 } else if (need_redraw(*old_state, *state)) {
69 ++global_counter<"widget:state:redraw">;
70 request_redraw();
71 }
72 }
73 old_state = *state;
74 });
75 }

324hi_inline widget_intf *get_if(widget_intf *start, widget_id id, bool include_invisible) noexcept
325{
326 hi_assert_not_null(start);
327
328 if (start->id == id) {
329 return start;
330 }
331 for (auto& child : start->children(include_invisible)) {
332 if (hilet r = get_if(&child, id, include_invisible); r != nullptr) {
333 return r;
334 }
335 }
336 return nullptr;
337}

339hi_inline widget_intf& get(widget_intf& start, widget_id id, bool include_invisible)
340{
341 if (auto r = get_if(std::addressof(start), id, include_invisible); r != nullptr) {
342 return *r;
343 }
344 throw not_found_error("get widget by id");
345}
